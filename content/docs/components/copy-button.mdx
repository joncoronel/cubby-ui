## API Reference

The CopyButton component extends the [Button component](/docs/components/button) with clipboard functionality. All Button props are supported except `size`, `variant`, `onClick`, and `children` which are controlled internally.

### Props

#### CopyButton

Button that copies text to clipboard with visual feedback. Extends the Button component.

<ApiPropsList>

<ApiProp name="content" fullType="string" required>
  The text content to copy to the clipboard when the button is clicked.
</ApiProp>

<ApiProp name="timeout" fullType="number" defaultValue="2000">
  Duration in milliseconds to show the success state (checkmark icon) after
  copying. Defaults to 2 seconds.
</ApiProp>

<ApiProp name="copyIcon" fullType="React.ReactNode" simpleType="node">
  Custom icon to display in the default (not copied) state. Defaults to the
  lucide Copy icon.
</ApiProp>

<ApiProp name="checkIcon" fullType="React.ReactNode" simpleType="node">
  Custom icon to display in the copied/success state. Defaults to the lucide
  Check icon with green color.
</ApiProp>

<ApiProp name="toast" fullType="true | Omit<AnchoredToastOptions, 'anchor'>" simpleType="boolean | object">
  Show an anchored toast above the button on successful copy. Pass `true` to
  use the defaults (`"Copied to clipboard!"`, `side: "top"`, arrow, duration
  matches `timeout`), or an options object to override any anchored toast
  option except `anchor` (which is always the button itself).
</ApiProp>

<ApiProp name="onCopied" fullType="(text: string) => void" simpleType="function">
  Fires after a successful copy with the copied text. Runs independently of
  `toast` — use it for analytics or extra side effects.
</ApiProp>

</ApiPropsList>
