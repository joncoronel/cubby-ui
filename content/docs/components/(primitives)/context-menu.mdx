import { ContextMenu, ContextMenuContent, ContextMenuItem, ContextMenuTrigger } from "@/registry/default/context-menu/context-menu";

The context menu opens on right-click and lists actions scoped to the element under the cursor. Use it for row actions in a table, canvas or editor commands, or any desktop-style secondary menu; for a button-triggered menu, use [`Dropdown Menu`](/docs/components/dropdown-menu) instead.

    ├── ContextMenuSeparator
    ├── ContextMenuShortcut
    └── ContextMenuSub
        ├── ContextMenuSubTrigger
        └── ContextMenuSubContent

## API Reference

### ContextMenuContent

<ApiPropsList>

<ApiProp
  name="align"
  fullType='"start" | "center" | "end"'
  simpleType="string"
  defaultValue='"start"'
>
  How to align the menu. This prop is flattened from the Positioner component for
  convenience.
</ApiProp>

<ApiProp name="sideOffset" fullType="number" defaultValue="4">
  Distance in pixels for positioning. This prop is flattened from the Positioner
  component for convenience.
</ApiProp>

<ApiProp name="level" fullType="1 | 2 | 3 | 4 | 5 | 6 | 7 | 8" defaultValue="3">
  Surface elevation level. See the [Surfaces](/docs/getting-started/surfaces) docs.
</ApiProp>

<ApiProp name="shadowLevel" fullType="1 | 2 | 3 | 4 | 5 | 6 | 7 | 8" defaultValue="3">
  Shadow weight. Controls drop-shadow intensity independently of substrate color.
</ApiProp>

</ApiPropsList>

### ContextMenuItem

<ApiPropsList>

<ApiProp name="inset" fullType="boolean">
  Add left padding to align with items that have icons, creating consistent
  visual alignment.
</ApiProp>

<ApiProp
  name="variant"
  fullType='"default" | "destructive"'
  simpleType="string"
  defaultValue='"default"'
>
  Visual style variant. Use 'destructive' for dangerous actions like delete or
  remove operations.
</ApiProp>

</ApiPropsList>

| Value         | Description                                       |
| ------------- | ------------------------------------------------ |
| `default`     | A standard menu action.                          |
| `destructive` | A dangerous action such as delete or remove.     |

### ContextMenuSubContent

<ApiPropsList>

<ApiProp name="sideOffset" fullType="number" defaultValue="0">
  Distance in pixels for positioning the submenu relative to its trigger.
</ApiProp>

<ApiProp
  name="align"
  fullType='"start" | "center" | "end"'
  simpleType="string"
  defaultValue='"start"'
>
  How to align the submenu relative to its trigger.
</ApiProp>

<ApiProp name="level" fullType="1 | 2 | 3 | 4 | 5 | 6 | 7 | 8" defaultValue="5">
  Surface elevation level. Submenus default to `5` so they pop above the parent menu at level 3.
</ApiProp>

<ApiProp name="shadowLevel" fullType="1 | 2 | 3 | 4 | 5 | 6 | 7 | 8" defaultValue="3">
  Shadow weight. Kept at `3` since submenus are still small popups.
</ApiProp>

</ApiPropsList>

### Additional Props

ContextMenu components inherit all props from [Base UI Context Menu](https://base-ui.com/react/components/context-menu#api-reference), including open, defaultOpen, onOpenChange, positioning options, and accessibility props.
