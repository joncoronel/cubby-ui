import { Collapsible, CollapsibleContent, CollapsibleTrigger } from "@/registry/default/collapsible/collapsible";

The collapsible shows or hides a single region of content behind a trigger, with a height animation. Use it for "show more" toggles, optional form sections, or sidebar groups; for a stacked set of toggling sections, use [`Accordion`](/docs/components/accordion) instead.
