import { Table, TableBody, TableCaption, TableCell, TableFooter, TableHead, TableHeader, TableRow } from "@/registry/default/table/table";

A table presents rows and columns of structured data inside a scrollable, card-styled frame. Use it for records you compare across columns, like users, invoices, or transactions, where each row shares the same fields.

## Composition

```text
Table
├── TableCaption
├── TableHeader
│   └── TableRow
│       └── TableHead
├── TableBody
│   └── TableRow
│       └── TableCell
└── TableFooter
    └── TableRow
        └── TableCell
```

`Table` reads its variant props (`bordered`, `striped`, and the rest) and passes them to the cells via data attributes, so the styling flags live on the root.

## API Reference

### Props

#### Table

Root container that wraps the table and provides variant context to children via data attributes. Features a card-like appearance with a muted background border.

<ApiPropsList>

<ApiProp
  name="bordered"
  fullType="boolean"
  defaultValue="false"
>
  Adds borders to all table cells for a grid appearance.
</ApiProp>

<ApiProp
  name="striped"
  fullType="boolean"
  defaultValue="false"
>
  Adds alternating background colors to rows for better readability.
</ApiProp>

<ApiProp
  name="hoverable"
  fullType="boolean"
  defaultValue="true"
>
  Enables hover effect on rows.
</ApiProp>

<ApiProp
  name="rowDividers"
  fullType="boolean"
  defaultValue="true"
>
  Shows horizontal dividers between body rows.
</ApiProp>

<ApiProp
  name="fadeEdges"
  fullType="boolean | 'x' | 'y' | 'top' | 'bottom' | 'left' | 'right'"
  defaultValue="'bottom'"
>
  Adds gradient fade effect at scroll edges to indicate more content.
</ApiProp>

<ApiProp
  name="nativeScroll"
  fullType="boolean"
  defaultValue="false"
>
  Uses native browser scrollbars instead of custom styled scrollbars.
</ApiProp>

<ApiProp
  name="hideScrollbar"
  fullType="boolean"
  defaultValue="false"
>
  Hides the scrollbar while maintaining scroll functionality.
</ApiProp>

<ApiProp
  name="persistScrollbar"
  fullType="boolean"
  defaultValue="false"
>
  Always shows the scrollbar instead of auto-hiding.
</ApiProp>

<ApiProp
  name="scrollbarGutter"
  fullType="boolean"
  defaultValue="false"
>
  Reserves space for scrollbar to prevent layout shift.
</ApiProp>

<ApiProp name="level" fullType="1 | 2 | 3 | 4 | 5 | 6 | 7 | 8" defaultValue="3">
  Surface elevation level for the Table's outer frame. See the [Surfaces](/docs/getting-started/surfaces) docs. The outer is a `bg-muted` gray frame; body cells drop to `bg-surface-3 dark:bg-surface-1` to form the inner card.
</ApiProp>

<ApiProp name="shadowLevel" fullType="1 | 2 | 3 | 4 | 5 | 6 | 7 | 8" defaultValue="1">
  Shadow weight for the outer frame. Defaults to `1`, a flat outlined card (rim only, no drops), matching the rest of the flat-card family.
</ApiProp>

</ApiPropsList>
