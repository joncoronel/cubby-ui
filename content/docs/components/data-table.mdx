## Column Definitions

Define columns using the `ColumnDef` type from TanStack Table. Each column can have:

- `accessorKey` - The key to access data from the row
- `header` - Header content (string or render function)
- `cell` - Cell content render function
- `enableSorting` - Whether the column is sortable (default: `true` when `enableSorting` prop is set)

When `enableSorting` is enabled, columns with string headers automatically render a sortable button with direction indicators. Use `meta.align` to control header alignment, or a render function for fully custom headers.

```tsx
import { type ColumnDef } from "@/registry/default/data-table/data-table"

const columns: ColumnDef<YourDataType>[] = [
  {
    accessorKey: "name",
    header: "Name", // Auto-renders with sort UI when enableSorting is true
  },
  {
    accessorKey: "amount",
    header: "Amount",
    meta: { align: "right" as const }, // Right-align the header
    cell: ({ row }) => {
      const amount = parseFloat(row.getValue("amount"))
      return <div className="text-right">${amount.toFixed(2)}</div>
    },
  },
  {
    accessorKey: "status",
    header: "Status",
    enableSorting: false, // Disable sorting for this column
  },
]
```

## API Reference

### Props

#### DataTable

A data-driven table component built on TanStack Table. Renders using the styled Table primitives.

<ApiPropsList>

<ApiProp
  name="columns"
  fullType="ColumnDef<TData, TValue>[]"
  required
>
  Array of column definitions. See TanStack Table documentation for full options.
</ApiProp>

<ApiProp
  name="data"
  fullType="TData[]"
  required
>
  Array of data objects to display in the table.
</ApiProp>

<ApiProp
  name="enableSorting"
  fullType="boolean"
  defaultValue="false"
>
  Enables column sorting. String headers automatically render with sort indicators. Set `enableSorting: false` on individual columns to disable sorting for that column.
</ApiProp>

<ApiProp
  name="enableRowSelection"
  fullType="boolean | ((row: Row<TData>) => boolean)"
  defaultValue="false"
>
  Enables row selection. Pass a function for conditional selection.
</ApiProp>

<ApiProp
  name="enableMultiRowSelection"
  fullType="boolean"
  defaultValue="true"
>
  Allows multiple rows to be selected. Set to `false` for single selection (radio button behavior).
</ApiProp>

<ApiProp
  name="showSelectionColumn"
  fullType="boolean"
  defaultValue="true"
>
  When `enableRowSelection` is true, automatically adds a checkbox column. Set to `false` to handle selection UI yourself.
</ApiProp>

<ApiProp
  name="enableFiltering"
  fullType="boolean"
  defaultValue="false"
>
  Enables column filtering.
</ApiProp>

<ApiProp

  fullType="boolean"
  defaultValue="false"
>
  Enables pagination.
</ApiProp>

<ApiProp
  name="enableColumnVisibility"
  fullType="boolean"
  defaultValue="false"
>
  Enables column visibility toggle. When enabled, a button appears in the toolbar that opens a popover to show/hide columns.
</ApiProp>

<ApiProp
  name="columnVisibility"
  fullType="VisibilityState"

  Controlled column visibility state. Use with `onColumnVisibilityChange` for controlled mode.
</ApiProp>

<ApiProp
