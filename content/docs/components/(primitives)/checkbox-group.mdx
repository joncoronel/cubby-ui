import { CheckboxGroup } from "@/registry/default/checkbox-group/checkbox-group";

The checkbox group manages the shared state of several related [`Checkbox`](/docs/components/checkbox) controls and wires up a parent checkbox for select-all behavior. Use it for multi-select preference lists, permission sets, or any group where selections combine into one value.
